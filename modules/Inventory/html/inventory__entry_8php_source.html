245 {

248  $cart->reference = $_POST['ref'];
249 
250  $cart->Comments = $_POST['Comments'];

356  display_error(_("The entered date is invalid."));

358  return false;
359  }

361  display_error(_("The entered date is not in fiscal year"));
362  set_focus('OrderDate');

366  display_error(_("You must enter at least one non empty item line."));
367  set_focus('AddItem');
368  return false;
369  }
370 
371  if (!$SysPrefs->allow_negative_stock() && ($low_stock = $_SESSION['Items']->check_qoh()))
372  {
373  display_error(_("This document cannot be processed because there is insufficient quantity for items marked."));
374  return false;
375  }
376 
377  if ($_SESSION['Items']->payment_terms['cash_sale'] == 0) {
378 
379  if (strlen($_POST['deliver_to']) <= 1) {
380  display_error(_("You must enter the person or company to whom delivery should be made to."));
381  set_focus('deliver_to');
382  return false;
383  }
384  if ($_SESSION['Items']->trans_type != ST_Inventory && strlen($_POST['delivery_address']) <= 1) {
385  display_error( _("You should enter the street address in the box provided. Orders cannot be accepted without a valid street address."));
386  set_focus('delivery_address');
387  return false;
388  }
389 
390  if ($_POST['freight_cost'] == "")
391  $_POST['freight_cost'] = price_format(0);
392 
393  if (!check_num('freight_cost',0)) {
394  display_error(_("The shipping cost entered is expected to be numeric."));
395  set_focus('freight_cost');
396  return false;
397  }
398  if (!is_date($_POST['delivery_date'])) {
399  if ($_SESSION['Items']->trans_type==ST_Inventory)
400  display_error(_("The Valid date is invalid."));
401  else
402  display_error(_("The delivery date is invalid."));
403  set_focus('delivery_date');
404  return false;
405  }
406  if (date1_greater_date2($_POST['OrderDate'], $_POST['delivery_date'])) {
407  if ($_SESSION['Items']->trans_type==ST_Inventory)
408  display_error(_("The requested valid date is before the date of the quotation."));
409  else
410  display_error(_("The requested delivery date is before the date of the order."));
411  set_focus('delivery_date');
412  return false;
413  }
414  }
415  else
416  {
417  if (!db_has_cash_accounts())
418  {
419  display_error(_("You need to define a cash account for your Sales Point."));
420  return false;
421  }
422  }
423  if (!$Refs->is_valid($_POST['ref'])) {
424  display_error(_("You must enter a reference."));
425  set_focus('ref');
426  return false;
427  }
428  if (!db_has_currency_rates($_SESSION['Items']->customer_currency, $_POST['OrderDate']))
429  return false;
430 
431  if ($_SESSION['Items']->get_items_total() < 0) {

492 function check_item_data()
493 {
494  global $SysPrefs, $allow_negative_prices;
495 
496  $is_inventory_item = is_inventory_item(get_post('stock_id'));
497  if(!get_post('stock_id_text', true)) {
498  display_error( _("Item description cannot be empty."));
499  set_focus('stock_id_edit');
500  return false;
501  }
502  elseif (!check_num('qty', 0) || !check_num('Disc', 0, 100)) {
503  display_error( _("The item could not be updated because you are attempting to set the quantity ordered to less than 0, or the discount percent to more than 100."));
504  set_focus('qty');
505  return false;
506  } elseif (!check_num('price', 0) && (!$allow_negative_prices || $is_inventory_item)) {
507  display_error( _("Price for inventory item must be entered and can not be less than 0"));
508  set_focus('price');
509  return false;
510  } elseif (isset($_POST['LineNo']) && isset($_SESSION['Items']->line_items[$_POST['LineNo']])
511  && !check_num('qty', $_SESSION['Items']->line_items[$_POST['LineNo']]->qty_done)) {
512 
513  set_focus('qty');
514  display_error(_("You attempting to make the quantity ordered a quantity less than has already been delivered. The quantity delivered cannot be modified retrospectively."));
515  return false;
516  }
517 
518  $cost_home = get_standard_cost(get_post('stock_id')); // Added 2011-03-27 Joe Hunt
519  $cost = $cost_home / get_exchange_rate_from_home_currency($_SESSION['Items']->customer_currency, $_SESSION['Items']->document_date);
520  if (input_num('price') < $cost)
521  {
522  $dec = user_price_dec();
523  $curr = $_SESSION['Items']->customer_currency;
524  $price = number_format2(input_num('price'), $dec);
525  if ($cost_home == $cost)
526  $std_cost = number_format2($cost_home, $dec);
527  else
528  {
529  $price = $curr . " " . $price;
530  $std_cost = $curr . " " . number_format2($cost, $dec);
531  }
532  display_warning(sprintf(_("Price %s is below Standard Cost %s"), $price, $std_cost));
533  }
534  return true;
535 }
